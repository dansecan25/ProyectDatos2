78
79 if(!this->states->isEmpty()){
80 this->states->peek()->stateUpdate(this->dt); //peeks the top element of the stack
81 if(this->states->peek()->getQuit()){
82 //here goes something if the game is quitted, if return to mainscren
83 this->states->peek()->endState();
84 this->states->pop(); //pop instantly frees the memory stored in the current top
